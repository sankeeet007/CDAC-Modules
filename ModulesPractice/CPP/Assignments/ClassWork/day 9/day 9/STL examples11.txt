STL examples in C++ for vector, map, stack, and queue. 

These are commonly used Standard Template Library (STL) containers.

1. Vector Example

A vector is a dynamic array that can grow or shrink automatically.

Features

Stored in contiguous memory
Dynamic size

Random access using index


Example
#include <iostream>
#include <vector>
using namespace std;

int main() {

    vector<int> v;

    v.push_back(10);
    v.push_back(20);
    v.push_back(30);

    cout << "Vector elements:" << endl;

    for(int i=0;i<v.size();i++)
    {
        cout << v[i] << " ";
    }

    return 0;
}

Output

10 20 30


Important Functions
Function	       Description
push_back(element)	insert element
pop_back(element)	remove last element
size()	                number of elements
at(position)	        access element
clear()	                remove all elements
erase(position)         remove element at position

3. Stack Example

A stack works on LIFO (Last In First Out).

Example

Push: 10 20 30
Pop: 30 20 10
Example Program
#include <iostream>
#include <stack>
using namespace std;

int main() {

    stack<int> s;

    s.push(10);
    s.push(20);
    s.push(30);

    cout << "Top element: " << s.top() << endl;

    s.pop();

    cout << "After pop Top: " << s.top();

}

Output

Top element: 30
After pop Top: 20
Important Functions
Function	Description
push()	insert element
pop()	remove top element
top()	view top element
empty()	check stack empty
4. Queue Example

A queue works on FIFO (First In First Out).

Example

Insert: 10 20 30
Remove: 10 20 30
Example Program
#include <iostream>
#include <queue>
using namespace std;

int main() {

    queue<int> q;

    q.push(10);
    q.push(20);
    q.push(30);

    cout << "Front: " << q.front() << endl;
    cout << "Rear: " << q.back() << endl;
    
    cout<<"  
    if(!q.empty())
        cout<<q.pop();
    else
        cout<<"Queue is empty, cannot pop";

    
}

Output

Front: 10
Rear: 30
After pop Front: 20


Important Functions
Function	Description
push(data)	insert element
pop()		remove front
front()		first element
back()		last element
empty()		check empty


Quick Comparison
Container	Structure	Access
vector	Dynamic Array	Random
map	Key-Value	Sorted keys
stack	LIFO	Top only
queue	FIFO	Front/Rear

Small Real Example (Student Marks using vector + map)

#include <iostream>
#include <vector>
#include <map>
using namespace std;

int main() {

    vector<int> marks = {80, 70, 90};

    map<int,string> students;

    students[1] = "Amit";
    students[2] = "Neha";

    cout<<"Marks:"<<endl;

    for(int m:marks)
        cout<<m<<" ";

    cout<<"\nStudents:"<<endl;

    for(auto s:students)
        cout<<s.first<<" "<<s.second<<endl;
}
