Q1.Create a procedure PC1 that uses a cursor to display employee name and salary for all employees.

DELIMITER //
CREATE PROCEDURE PC1()
BEGIN
		-- declare local variables
		DECLARE v_ename VARCHAR(50);
		DECLARE v_sal INT;
		DECLARE done INT DEFAULT 0;
		
		-- cursor declaration
		DECLARE emp_cursor CURSOR FOR
		SELECT ename, sal
		FROM emp;
		
		-- handler 
		DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;
		
		-- Open cursor
		OPEN emp_cursor;
		
	read_loop : LOOP	-- loop 

		-- Fetch CURSOR		
		FETCH emp_cursor INTO v_ename, v_sal;
		
		-- checking if there are no more rows	
		IF done THEN
            LEAVE read_loop;
        END IF;
		
		SELECT v_ename, v_sal;
	END LOOP;
	CLOSE emp_cursor;
END //
DELIMITER ;


	CALL PC1;

Q8.Create a procedure PC8 that uses a cursor to print employee names along with experience in years from HIREDATE till today.

DELIMITER //
CREATE PROCEDURE PC8()
BEGIN
		-- declare local variables
		DECLARE v_ename VARCHAR(50);
		DECLARE v_experience INT;
		DECLARE v_hiredate DATE;
		DECLARE done INT DEFAULT 0;
		
		-- cursor declaration
		DECLARE emp_cursor CURSOR FOR
		SELECT ename, hiredate
		FROM emp;
		
		-- handler 
		DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;
		
		-- Open cursor
		OPEN emp_cursor;
		
	read_loop : LOOP	-- loop 

		-- Fetch CURSOR		
		FETCH emp_cursor INTO v_ename, v_hiredate;
		
		-- checking if there are no more rows	
		IF done THEN
            LEAVE read_loop;
        END IF;
		
		SET v_experience = TIMESTAMPDIFF(YEAR, v_hiredate, CURDATE());
		
		SELECT v_ename AS Emp_Name, v_experience AS Experience;
		
	END LOOP;
	
	CLOSE emp_cursor;
END //
DELIMITER ;
