Q1. Display the average, lowest, highest and difference in lowest and highest salaries for each department number.

	SELECT deptno, AVG(sal), MIN(SAL), MAX(SAL),MAX(SAL)-MIN(SAL) as "difference"
	FROM emp
	GROUP BY deptno
	ORDER BY deptno;
	
	
Q2. Display the number of employees within each job.

	SELECT job, count(empno) as "No of Emp"
	FROM emp
	GROUP BY job;
	
Q3. Display the lowest salaries job-wise within each department number.
		
	SELECT job, deptno, MIN(sal) as "minimum sals"
	FROM emp
	GROUP BY job, deptno
	ORDER BY deptno, job;

Q4. Display only those jobs where the minimum salary is greater than or equal to 3000.

	SELECT job, MIN(sal) as "Min sal"
	FROM emp
	GROUP BY job
	HAVING MIN(sal) >= 3000;
	
Q5. Display the records sorted as per the job. Make sure that within each job the records are sorted as per the highest to lowest salaries.

	SELECT job, sal
	FROM emp
	ORDER BY job, sal DESC;

Q6. Display Deptno, total of that deptno and Grade. Grade should be A if the total is greater than 10000, else it should be B. Sort the summary output as per the highest to lowest Grade.

	SELECT deptno, sum(sal) as "Total", case
	WHEN sum(sal) > 10000 THEN 'A' ELSE 'B'
	end as Grade
	FROM emp
	GROUP BY deptno
	ORDER BY deptno;
	
	
Q7. Calculate and display only the average salaries paid to all those employees where the employee name starts with A and name starts with S.

	SELECT ename, AVG(sal)
	FROM emp
	GROUP BY ename
	HAVING ename LIKE 'A%' OR ename LIKE 'S%';

Q10. Produce exactly same output as shown below:

Q23.Display the total amount as per code which will be grouped by 3rd, 4th, 5th and 6th characters of code column’s values.

    select code, sum(amount) as total_amount
    from code_data
    group by code
    having code like '__ABCD_____' or code like '__JKLM_____';
	
Q24.The last 4 digits of code values represent the year. Display year wise total amount.

    SELECT
    RIGHT(code, 4) AS year,
    SUM(amount) AS total_amount
    FROM code_data
    WHERE code LIKE '_______2022' OR code LIKE '_______2023'
    GROUP BY RIGHT(code, 4);
	
Q25.Display the average amount as per the 6th character of code column’s values. 

    SELECT
    SUBSTRING(code, 6, 1) AS sixth_char,
    AVG(amount) AS Average_Amount
    FROM code_data
    WHERE code LIKE '_____D%' OR code LIKE '_____M%'
    GROUP BY SUBSTRING(code, 6, 1);

Q14.Display transaction id, continent name, country name, city name, amount allocated and amount spent.

    SELECT
         T.TrID,
         Con.Name_of_Continent,
         Cou.Name_of_Country,
         T.Name_of_City,
         Cou.Country_Amount_Allocated,
         T.Amount_Spent
    FROM
         Transactions T
    JOIN
         country_master Cou ON T.CountryID = Cou.CountryID
    JOIN
         continent_master Con ON Cou.ContinentID = Con.ContinentID;
	
Q15.Display the total amount spent for each city, within each country from each continent

    SELECT
         Con.Name_of_Continent,
         Cou.Name_of_Country,
         T.Name_of_City,
         SUM(T.Amount_Spent) AS Total_City_Spent
    FROM
         Transactions T
    JOIN
         country_master Cou ON T.CountryID = Cou.CountryID
    JOIN
         continent_master Con ON Cou.ContinentID = Con.ContinentID
    GROUP BY
         Con.Name_of_Continent,
         Cou.Name_of_Country,
         T.Name_of_City
    ORDER BY
         Con.Name_of_Continent,
         Cou.Name_of_Country;

Q1. Display patient name, doctor name, and department for completed appointments.

	SELECT 	p.patientname as "Patient Name",
			doc.doctorname as "Doctor Name", 
			dept.departmentname as "Department Name",
			apt.appointmentstatus as "Appointment Status"
	FROM patients p
	JOIN appointments apt ON p.PatientID = apt.PatientID
	JOIN doctors doc ON apt.DoctorID = doc.DoctorID
	JOIN departments dept ON doc.DepartmentID = dept.DepartmentID
	WHERE apt.appointmentstatus = 'Completed'
	ORDER BY p.patientname;

Q2. List appointments in March 2024 with treatment cost > 600.

	SELECT 	p.patientname as "Patient Name",
			doc.doctorname as "Doctor Name", 
			apt.appointmentDate as "Appointment Date",
			apt.treatmentCost as "Treatment Cost"
	FROM patients p
	JOIN appointments apt ON p.patientID = apt.patientID
	JOIN doctors doc ON apt.doctorID = doc.doctorID
	WHERE apt.appointmentDate LIKE '_____03___'  AND apt.treatmentCost > 600
	ORDER BY p.patientname;
			
Q3. Show doctors from Cardiology department who handled more than 3 appointments.

    FROM Doctors d
    JOIN Departments dep ON d.DepartmentID = dep.DepartmentID
    JOIN Appointments a ON d.DoctorID = a.DoctorID
    WHERE dep.DepartmentName = 'Cardiology'
    GROUP BY d.DoctorID
    HAVING AppointmentCount > 3;
