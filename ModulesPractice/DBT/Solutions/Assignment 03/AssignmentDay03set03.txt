Q1. Display patient name, doctor name, and department for completed appointments.

	SELECT 	p.patientname as "Patient Name",
			doc.doctorname as "Doctor Name", 
			dept.departmentname as "Department Name",
			apt.appointmentstatus as "Appointment Status"
	FROM patients p
	JOIN appointments apt ON p.PatientID = apt.PatientID
	JOIN doctors doc ON apt.DoctorID = doc.DoctorID
	JOIN departments dept ON doc.DepartmentID = dept.DepartmentID
	WHERE apt.appointmentstatus = 'Completed'
	ORDER BY p.patientname;

Q2. List appointments in March 2024 with treatment cost > 600.

	SELECT 	p.patientname as "Patient Name",
			doc.doctorname as "Doctor Name", 
			apt.appointmentDate as "Appointment Date",
			apt.treatmentCost as "Treatment Cost"
	FROM patients p
	JOIN appointments apt ON p.patientID = apt.patientID
	JOIN doctors doc ON apt.doctorID = doc.doctorID
	WHERE apt.appointmentDate LIKE '_____03___'  AND apt.treatmentCost > 600
	ORDER BY p.patientname;
			
Q3. Show doctors from Cardiology department who handled more than 3 appointments.

	SELECT d.DoctorName, COUNT(a.AppointmentID) AS AppointmentCount
    FROM Doctors d
    JOIN Departments dep ON d.DepartmentID = dep.DepartmentID
    JOIN Appointments a ON d.DoctorID = a.DoctorID
    WHERE dep.DepartmentName = 'Cardiology'
    GROUP BY d.DoctorID
    HAVING AppointmentCount > 3;
	

Q4. Display total revenue generated per department.

SELECT dep.DepartmentName, SUM(a.TreatmentCost) AS TotalRevenue
    -> FROM Departments dep
    -> JOIN Doctors d ON dep.DepartmentID = d.DepartmentID
    -> JOIN Appointments a ON d.DoctorID = a.DoctorID
    -> GROUP BY dep.DepartmentName;

Q5. List patients who never had a cancelled appointment.

SELECT PatientName FROM Patients
    -> WHERE PatientID NOT IN (
    ->     SELECT PatientID FROM Appointments WHERE AppointmentStatus = 'Cancelled'
    -> );

Q6. Show doctors earning more than 140000 who handled completed appointments.

 SELECT DISTINCT d.DoctorName, d.Salary
    -> FROM Doctors d
    -> JOIN Appointments a ON d.DoctorID = a.DoctorID
    -> WHERE d.Salary > 140000 AND a.AppointmentStatus = 'Completed';

Q7. Display patients treated by doctors from Oncology department.

SELECT DISTINCT p.PatientName
    -> FROM Patients p
    -> JOIN Appointments a ON p.PatientID = a.PatientID
    -> JOIN Doctors d ON a.DoctorID = d.DoctorID
    -> JOIN Departments dep ON d.DepartmentID = dep.DepartmentID
    -> WHERE dep.DepartmentName = 'Oncology';

Q8. Show total appointments per month.

SELECT DATE_FORMAT(AppointmentDate, '%Y-%m') AS Month, COUNT(*) AS TotalAppointments
    -> FROM Appointments
    -> GROUP BY Month;

Q9. Find top 3 doctors based on total treatment revenue generated.

SELECT d.DoctorName, SUM(a.TreatmentCost) AS TotalRevenue
    -> FROM Doctors d
    -> JOIN Appointments a ON d.DoctorID = a.DoctorID
    -> GROUP BY d.DoctorID
    -> ORDER BY TotalRevenue DESC
    -> LIMIT 3;

Q10. Display patients from USA who had appointments in April 2024.

SELECT DISTINCT p.PatientName
    -> FROM Patients p
    -> JOIN Appointments a ON p.PatientID = a.PatientID
    -> WHERE p.Country = 'USA'
    -> AND a.AppointmentDate BETWEEN '2024-04-01' AND '2024-04-30';

Q11. List departments that had no cancelled appointments.

SELECT DepartmentName FROM Departments
    -> WHERE DepartmentID NOT IN (
    ->     SELECT d.DepartmentID
    ->     FROM Doctors d
    ->     JOIN Appointments a ON d.DoctorID = a.DoctorID
    ->     WHERE a.AppointmentStatus = 'Cancelled'
    -> );

Q12. Show doctors who handled more than 5 completed appointments.

SELECT d.DoctorName, COUNT(a.AppointmentID) AS CompletedCount
    -> FROM Doctors d
    -> JOIN Appointments a ON d.DoctorID = a.DoctorID
    -> WHERE a.AppointmentStatus = 'Completed'

Q13. Display total revenue generated by each doctor.

SELECT d.DoctorName, SUM(a.TreatmentCost) AS Revenue
    -> FROM Doctors d
    -> LEFT JOIN Appointments a ON d.DoctorID = a.DoctorID
    -> GROUP BY d.DoctorID;

Q14. List patients who visited more than 2 different departments.

SELECT p.PatientName
    -> FROM Patients p
    -> JOIN Appointments a ON p.PatientID = a.PatientID
    -> JOIN Doctors d ON a.DoctorID = d.DoctorID
    -> GROUP BY p.PatientID
    -> HAVING COUNT(DISTINCT d.DepartmentID) > 2;

Q15. Show appointments where treatment cost is greater than average treatment cost.

 SELECT * FROM Appointments

Q16. Display department-wise average treatment cost.

SELECT dep.DepartmentName, AVG(a.TreatmentCost) AS AvgCost
    -> FROM Departments dep
    -> JOIN Doctors d ON dep.DepartmentID = d.DepartmentID
    -> JOIN Appointments a ON d.DoctorID = a.DoctorID
    -> GROUP BY dep.DepartmentName;

Q17. List doctors who handled appointments in every month from Jan to June 2024.

SELECT d.DoctorName
    -> FROM Doctors d
    -> JOIN Appointments a ON d.DoctorID = a.DoctorID
    -> WHERE a.AppointmentDate BETWEEN '2024-01-01' AND '2024-06-30'
    -> GROUP BY d.DoctorID
    -> HAVING COUNT(DISTINCT MONTH(a.AppointmentDate)) = 6;

Q18. Show patients who had both Scheduled and Completed appointments.

 SELECT p.PatientName
    -> FROM Patients p
    -> JOIN Appointments a ON p.PatientID = a.PatientID
    -> WHERE a.AppointmentStatus IN ('Scheduled', 'Completed')
    -> GROUP BY p.PatientID
    -> HAVING COUNT(DISTINCT a.AppointmentStatus) = 2;
