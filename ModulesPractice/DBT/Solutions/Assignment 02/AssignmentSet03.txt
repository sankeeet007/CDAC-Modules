Q1. Update GPA to 3.00 for all students whose GPA is below 2.00.

UPDATE students 
SET GPA = 3.00 
WHERE GPA < 2.00;

Q2 Increase salary by 8% for instructors in the 'Mathematics' department.

UPDATE instructors
SET salary = salary + (salary*0.08)
WHERE department = 'Mathematics';

Q3 Set student status to 'INACTIVE' for students registered before 2021.

UPDATE students 
SET status = 'INACTIVE'
WHERE year(registration_date) < 2021;

Q4 Reduce course fee by 10% for courses having credits = 3.

UPDATE courses 
SET fee = fee-(fee*0.1)
WHERE credits = 3;

Q5 Update payment_status to 'PAID' where grade is not blank.

UPDATE enrollments
SET payment_status = 'PAID'
WHERE grade IS NOT NULL;

Q6 Increase GPA by 0.10 for students from Texas.

UPDATE students
SET GPA = GPA + 0.10
WHERE state = 'Texas';

Q7 Update fee to 1000 for 'Web Development' course.

UPDATE courses 
SET fee = 1000
WHERE course_name = 'Web Development';

Q8 Delete enrolments which have grade as 'F'.

DELETE FROM enrollments
WHERE grade = 'F';

Q9 Increase salary by 5% for instructors whose salary is below 70000.

UPDATE instructors
SET salary = salary + (salary*0.05)
WHERE salary < 70000;

Q10 Delete records from enrolments table of students who have not paid.

DELETE FROM enrollments
WHERE payment_status = 'UNPAID';
