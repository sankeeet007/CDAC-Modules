Conditional ANSI SQL Function "Case".
Case acting as Switch Case (exactly value comparison)

select ename, deptno, case deptno
                           when 10 then 'A'
			   when 20 then 'B'
			else
			'C'
		       end as "Deptno Code"
from emp;

select ename, deptno, case deptno
                           when 10 then 'A'
			   when 20 then 'B'

		       end as "Deptno Code"
from emp;

Case acting as if ... else if .. else (based on a boolean condition)

select ename, sal, case 
			when sal >= 2900 then 'A'
			when sal >= 2600 then 'B'
			when sal >= 2000 then 'C'
			else
			'D'
		   end as  "Salary Class"
from emp;
